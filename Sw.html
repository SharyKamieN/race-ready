// Race Ready — Service Worker const CACHE = 'race-ready-v1'; self.addEventListener('install', e => { self.skipWaiting(); }); self.addEventListener('activate', e => { e.waitUntil(clients.claim()); }); // Network first — always fresh data, fallback to cache self.addEventListener('fetch', e => { // Never cache SSE events or API calls if (e.request.url.includes('/events') || e.request.url.includes('/api/') || e.request.method !== 'GET') { return; } e.respondWith( fetch(e.request) .then(res => { if (res.ok) { const clone = res.clone(); caches.open(CACHE).then(c => c.put(e.request, clone)); } return res; }) .catch(() => caches.match(e.request)) ); });
